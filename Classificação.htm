  <script>

    // Adiciona opções às listas de seleção

    const opcoes = [

      "0 = Não me descreve de forma alguma",

      "1 = Me descreve muito pouco",

      "2 = Me descreve um pouco",

      "3 = Me descreve razoavelmente",

      "4 = Me descreve bem",

      "5 = Me descreve muito bem",

      "6 = Me descreve totalmente"

    ];

    function carregarOpcoes() {

      for (let i = 1; i <= 6; i++) {

        let sel = document.getElementById("DG" + i);

        opcoes.forEach((texto, idx) => {

          let opt = document.createElement("option");

          opt.value = idx;

          opt.text = texto;

          sel.appendChild(opt);

        });

      }

      for (let i = 1; i <= 8; i++) {

        let sel = document.getElementById("SW" + i);

        opcoes.forEach((texto, idx) => {

          let opt = document.createElement("option");

          opt.value = idx;

          opt.text = texto;

          sel.appendChild(opt);

        });

      }

    }

 

    function inverterSW2(valor) {

      const mapa = {0:6, 1:5, 2:4, 3:3, 4:2, 5:1, 6:0};

      return mapa[valor];

    }

 

    function classificar() {

      const dgdi = [];

      for (let i = 1; i <= 6; i++) {

        dgdi.push(parseInt(document.getElementById("DG" + i).value));

      }

      const mediaDGDI = dgdi.reduce((a,b) => a+b, 0) / dgdi.length;

 

      const swdb = [];

      for (let i = 1; i <= 8; i++) {

        let val = parseInt(document.getElementById("SW" + i).value);

        if (i === 2) val = inverterSW2(val);

        swdb.push(val);

      }

      const mediaSWDB = swdb.reduce((a,b) => a+b, 0) / swdb.length;

 

      let grupo = "Não Elegível";

      if (mediaDGDI <= 1.5 && mediaSWDB >= 4.5) grupo = "Grupo G1";

      else if (mediaDGDI >= 4.5 && mediaSWDB <= 1.5) grupo = "Grupo G2";

 

      document.getElementById("resultado").innerText =

        `🧮 Média DG-DI: ${mediaDGDI.toFixed(2)}\n🧮 Média SWDB: ${mediaSWDB.toFixed(2)}\n\n✅ Classificação: ${grupo}`;

    }

 

    carregarOpcoes();

  </script>
